@Composable
fun ListScreen(
    viewModel: ListViewModel = viewModel(),
    modifier: Modifier = Modifier,
    // idle timeout in milliseconds; tests can pass a smaller value
    idleTimeoutMs: Long = 30_000L,
    // action to run when idle detected; default logs a message
    onIdle: () -> Unit = { Log.i("IdlePrinter", "Idle for $idleTimeoutMs ms - printing...") }
) {
    val state by viewModel.uiState.collectAsState()

    // Verify composable is composed at startup
    Log.e("DBG", "ListScreen entered")

    // track the last time the user interacted (click/tap/scroll will update)
    val lastInteractionTimestamp = remember { mutableStateOf(System.currentTimeMillis()) }
    fun recordInteraction() {
        lastInteractionTimestamp.value = System.currentTimeMillis()
        // Log the actual long value (not the MutableState object) so it appears clearly in Logcat
        Log.i("lastInteractionTimestamp", "lastInteractionTimestamp ${lastInteractionTimestamp.value} ms")
        // Extra high-priority log to make verification easier while debugging
        Log.e("DBG", "recordInteraction() called — ${lastInteractionTimestamp.value}")
    }

    // Full-screen pointer input that records any user interaction (touch/drag/etc.)
    // so we don't need to add recordInteraction() to every clickable/radio button.
    val fullScreenInteractionModifier = Modifier.pointerInteropFilter {
        // any MotionEvent -> record and allow propagation
        recordInteraction()
        false
    }

    Surface(modifier = modifier.fillMaxSize().then(fullScreenInteractionModifier)) {

        val options = listOf("All","Reroute", "Return", "Forward")
        var selectedOption by remember { mutableStateOf(options[0]) }


        val sortOptions = listOf("Ascending", "Descending")
        var sortSelectedOption by remember { mutableStateOf(sortOptions[0]) }

        // remember a LazyListState to control scroll position
        val listState = rememberLazyListState()

        // Coroutine that checks for idle state periodically and triggers onIdle once per idle period
        LaunchedEffect(Unit) {
            while (true) {
                delay(500)
                val idleTime = System.currentTimeMillis() - lastInteractionTimestamp.value
                if (idleTime >= idleTimeoutMs && !listState.isScrollInProgress) {
                    try {
                        onIdle()
                    } catch (t: Throwable) {
                        Log.e("IdlePrinter", "onIdle failed", t)
                    }
                    // update timestamp so we don't repeatedly fire until the next interaction
                    lastInteractionTimestamp.value = System.currentTimeMillis()
                }
            }
        }

        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(16.dp),
        )
        {
            Text("Sample List View", style = MaterialTheme.typography.titleLarge,
                modifier = Modifier
                    .fillMaxWidth()
                    .padding(16.dp, top = 50.dp, bottom = 50.dp))
            // Use a LazyRow so options are arranged horizontally and scroll when needed
            LazyRow(
                modifier = Modifier
                    .fillMaxWidth()
                    .padding(bottom = 8.dp),
                horizontalArrangement = Arrangement.spacedBy(8.dp),
                contentPadding = PaddingValues(horizontal = 8.dp)
            ) {
                items(options) { option ->
                    Row(
                        modifier = Modifier
                            .clickable { selectedOption = option }
                            .padding(8.dp)
                    ) {
                        RadioButton(
                            selected = (option == selectedOption),
                            onClick = { selectedOption = option }
                        )
                        Text(text = option)
                    }
                }
            }

            LazyRow(
                modifier = Modifier
                    .fillMaxWidth()
                    .padding(bottom = 8.dp),
                horizontalArrangement = Arrangement.spacedBy(8.dp),
                contentPadding = PaddingValues(horizontal = 8.dp)
            ) {
                items(sortOptions) { option ->
                    Row(
                        modifier = Modifier
                            .clickable { sortSelectedOption = option }
                            .padding(8.dp)
                    ) {
                        RadioButton(
                            selected = (option == sortSelectedOption),
                            onClick = { sortSelectedOption = option }
                        )
                        Text(text = option)
                    }
                }
            }

            // Memoize filtered+sorted result so it recomputes only when inputs change
            val sorted by remember(selectedOption, sortSelectedOption, state.items) {
                derivedStateOf {
                    val filtered = if (selectedOption == "All") state.items else state.items.filter { it.title == selectedOption }
                    when (sortSelectedOption) {
                        "Ascending" -> filtered.sortedBy { it.bagTagId }
                        "Descending" -> filtered.sortedByDescending { it.bagTagId }
                        else -> filtered
                    }
                }
            }

            // Scroll to first item whenever the filter or sort selection changes
            LaunchedEffect(selectedOption, sortSelectedOption) {
                if (sorted.isNotEmpty()) {
                    listState.scrollToItem(0)
                }
            }

            LazyColumn(
                state = listState,
                modifier = Modifier
                    .fillMaxWidth()
                    .padding(16.dp),
                verticalArrangement = Arrangement.spacedBy(8.dp),
                contentPadding = PaddingValues(bottom = 16.dp)
            ) {
                items(sorted, key = { it.bagTagId }) { item ->
                    // root full-screen pointer listener handles touches, so omit per-item callback
                    ItemRow(item)
                }
            }
        }
    }
}

@Composable
fun ItemRow(item: Item, onInteract: () -> Unit = {}) {
    Card(
        modifier = Modifier
            .fillMaxWidth()
            .clickable { onInteract()},
        shape = RoundedCornerShape(8.dp),
        elevation = CardDefaults.cardElevation(defaultElevation = 4.dp)
    ) {
        Column(Modifier.padding(16.dp)) {
            Text(text = item.bagTagId.toString(), fontWeight = FontWeight.Bold, fontSize = 18.sp)
        }
    }
}
